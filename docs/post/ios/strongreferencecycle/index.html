import UIKit

class ViewController: UIViewController {
    var b = B()
    
    override func viewDidLoad() {
        super.viewDidLoad()
                
        let a = A(b: self.b)
        self.b.a = a
    }
}

class A {
    let b: B
    
    init(b: B) {
        self.b = b
    }
}

class B {
    var a: A?
    
    init() {
        self.a = nil
    }
}

From the memory graph, it’s pretty clear that object A and B have strong reference cycle.

import UIKit

class ViewController: UIViewController {
        
    override func viewDidLoad() {
        super.viewDidLoad()
        
        let heading = HTMLElement(name: "h1")
        print(heading.asHTML())
    }
}

class HTMLElement {

    let name: String
    let text: String?

    lazy var asHTML: () -> String = {
        if let text = self.text {
            return "<\(self.name)>\(text)</\(self.name)>"
        } else {
            return "<\(self.name) />"
        }
    }

    init(name: String, text: String? = nil) {
        self.name = name
        self.text = text
        
        print("\(name) is being initialized")
    }

    deinit {
        print("\(name) is being deinitialized")
    }

}

But if you change lazy var asHTML: () -> String to lazy var asHTML: String and the code accordingly, the strong reference cycle won’t happen!
